function Show-Tree {
    param ($path = ".", $prefix = "")

    Get-ChildItem $path | ForEach-Object {
        if ($_ -is [System.IO.DirectoryInfo] -and ($exclude -notcontains $_.Name)) {
            Write-Output "$prefix|-- $_"
            Show-Tree "$($_.FullName)" "$prefix|   "
        } elseif ($_ -is [System.IO.FileInfo]) {
            Write-Output "$prefix|-- $_"
        }
    }
}

$exclude = @("build", "dist", "venv", "__pycache__","api")
